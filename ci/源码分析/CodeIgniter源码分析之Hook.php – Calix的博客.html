CodeIgniter源码分析之Hook.php

CodeIgniter 2年前 (2014-11-19) 699℃

作者：Calix

  1. <?php if ( ! defined('BASEPATH')) exit('No direct script access allowed');
  2. // ------------------------------------------------------------------------
  3.  
  4. /**
  5. * CodeIgniter Hooks Class
  6. */
  7. class CI_Hooks {
  8.  
  9. /**
  10. * Determines wether hooks are enabled
  11. */
  12. var $enabled = FALSE;
  13.  
  14. /**
  15. * List of all hooks set in config/hooks.php
  16. */
  17. var $hooks = array();
  18.  
  19. /**
  20. * Determines wether hook is in progress, used to prevent infinte loops
  21. */
  22. var $in_progress = FALSE;
  23.  
  24. /**
  25. * Constructor
  26. */
  27. function __construct()
  28. {
  29. $this->_initialize();
  30. log_message('debug', "Hooks Class Initialized");
  31. }
  32.  
  33. // --------------------------------------------------------------------
  34.  
  35. /**
  36. * Initialize the Hooks Preferences
  37. */
  38. function _initialize()
  39. {
  40. $CFG =& load_class('Config', 'core');
  41.  
  42. //如果配置文件中设置了是不允许hooks，则直接返回退出本函数。
  43. if ($CFG->item('enable_hooks') == FALSE)
  44. {
  45. return;
  46. }
  47.  
  48. //要使用到的钩子，必须在配置目录下的hooks.php里面定义好。否则无法使用。
  49. if (defined('ENVIRONMENT') AND is_file(APPPATH.'config/'.ENVIRONMENT.'/hooks.php'))
  50. {
  51. include(APPPATH.'config/'.ENVIRONMENT.'/hooks.php');
  52. }
  53. elseif (is_file(APPPATH.'config/hooks.php'))
  54. {
  55. include(APPPATH.'config/hooks.php');
  56. }
  57.  
  58. if ( ! isset($hook) OR ! is_array($hook))
  59. {
  60. return;
  61. }
  62.  
  63. //把钩子信息都保存到Hook组件中。
  64. $this->hooks =& $hook;
  65. $this->enabled = TRUE;
  66. }
  67.  
  68. // --------------------------------------------------------------------
  69.  
  70. /**
  71. * Call Hook
  72. * 外部其实就是调用这个_call_hook函数进行调用钩子程序。而此方法中再调用_run_hook去执行相应的钩子。
  73. */
  74. function _call_hook($which = '')
  75. {
  76. if ( ! $this->enabled OR ! isset($this->hooks[$which]))
  77. {
  78. return FALSE;
  79. }
  80.  
  81. //同一个位置可以执行多个hook
  82. if (isset($this->hooks[$which][0]) AND is_array($this->hooks[$which][0]))
  83. {
  84. foreach ($this->hooks[$which] as $val)
  85. {
  86. $this->_run_hook($val);
  87. }
  88. }
  89. else
  90. {
  91. $this->_run_hook($this->hooks[$which]);
  92. }
  93.  
  94. return TRUE;
  95. }
  96.  
  97. // --------------------------------------------------------------------
  98.  
  99. /**
  100. * Run Hook
  101. */
  102. function _run_hook($data)
  103. {
  104. //一般来说，这个$data会有：类名，方法名，参数，类文件路径等参数。
  105. if ( ! is_array($data))
  106. {
  107. return FALSE;
  108. }
  109.  
  110. // -----------------------------------
  111. // Safety - Prevents run-away loops
  112. // -----------------------------------
  113.  
  114. //如果调用某一个hook，执行某些脚本，而有可能这些脚本里面再会触发其它hook，如果这个其它hook里面又包含了当前
  115. //的hook，那么就会进入死循环，这个in_progress的存在就是阻止这种情况。
  116. if ($this->in_progress == TRUE)
  117. {
  118. return;
  119. }
  120.  
  121. //下面都是一些执行钩子的预处理，包括判断类文件是否存在，类和方法是否正确等等。
  122. // -----------------------------------
  123. // Set file path
  124. // -----------------------------------
  125.  
  126. if ( ! isset($data['filepath']) OR ! isset($data['filename']))
  127. {
  128. return FALSE;
  129. }
  130.  
  131. $filepath = APPPATH.$data['filepath'].'/'.$data['filename'];
  132.  
  133. if ( ! file_exists($filepath))
  134. {
  135. return FALSE;
  136. }
  137.  
  138. // -----------------------------------
  139. // Set class/function name
  140. // -----------------------------------
  141.  
  142. $class = FALSE;
  143. $function = FALSE;
  144. $params = '';
  145.  
  146. if (isset($data['class']) AND $data['class'] != '')
  147. {
  148. $class = $data['class'];
  149. }
  150.  
  151. if (isset($data['function']))
  152. {
  153. $function = $data['function'];
  154. }
  155.  
  156. if (isset($data['params']))
  157. {
  158. $params = $data['params'];
  159. }
  160.  
  161. if ($class === FALSE AND $function === FALSE)
  162. {
  163. return FALSE;
  164. }
  165.  
  166. // -----------------------------------
  167. // Set the in_progress flag
  168. // -----------------------------------
  169.  
  170. //在开始执行钩子相应的程序之前，先把当前hook的状态设为正在运行中。
  171. $this->in_progress = TRUE;
  172.  
  173. // -----------------------------------
  174. // Call the requested class and/or function
  175. // -----------------------------------
  176.  
  177. //执行
  178. if ($class !== FALSE)
  179. {
  180. if ( ! class_exists($class))
  181. {
  182. require($filepath);
  183. }
  184.  
  185. $HOOK = new $class;
  186. $HOOK->$function($params);
  187. }
  188. else
  189. {
  190. if ( ! function_exists($function))
  191. {
  192. require($filepath);
  193. }
  194.  
  195. $function($params);
  196. }
  197.  
  198. //执行相应程序完毕后，重新把当前hook的状态改为非运行中，以让它可以再次被触发。
  199. $this->in_progress = FALSE;
  200. return TRUE;
  201. }
  202.  
  203. }
